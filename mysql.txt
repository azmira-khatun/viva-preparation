üîπ Data Read / Query (SELECT / Filtering / Sorting / Limit)
-- ‡¶∏‡¶¨ column ‡¶∏‡¶π ‡¶∏‡¶¨ rows select ‡¶ï‡¶∞‡¶§‡ßá

-- ‡¶®‡¶ø‡¶∞‡ßç‡¶¶‡¶ø‡¶∑‡ßç‡¶ü column ‡¶ó‡ßÅ‡¶≤‡ßã select
SELECT col1, col2 FROM table_name;

-- row filter ‡¶ï‡¶∞‡¶§‡ßá (condition ‡¶Ö‡¶®‡ßÅ‡¶Ø‡¶æ‡ßü‡ßÄ)
SELECT * FROM table_name WHERE col1 = 'some value';

-- multiple condition ‡¶π‡¶≤‡ßá AND / OR

WHERE col1 = 'value' AND col2 > 10;

-- range‚Äë‡¶è‡¶∞ ‡¶≠‡¶ø‡¶§‡¶∞ filter (numeric / date etc.)

WHERE col BETWEEN value1 AND value2;

-- ‡¶≤‡¶ø‡¶∏‡ßç‡¶ü‚Äë‡¶≠‡ßÅ‡¶ï‡ßç‡¶§ ‡¶ï‡ßã‡¶® ‡¶Æ‡¶æ‡¶® ‡¶™‡¶æ‡¶ì‡ßü‡¶æ ‡¶Ø‡¶æ‡¶¨‡ßá ‡¶ï‡¶ø ‡¶®‡¶æ ‡¶ö‡ßá‡¶ï ‡¶ï‡¶∞‡¶§‡ßá

WHERE col IN (value1, value2, value3);

-- pattern matching (text search) ‚Äî ‡¶Ø‡ßá‡¶Æ‡¶®: starts with, contains
SELECT * FROM table_name
WHERE col LIKE 'abc%';  -- starts with 'abc'
SELECT * FROM table_name
WHERE col LIKE '%xyz%'; -- contains 'xyz'

-- order by, sorting
SELECT * FROM table_name
ORDER BY col1 ASC;   -- ascending
SELECT * FROM table_name
ORDER BY col2 DESC;  -- descending

-- limit + offset (pagination / top‚Äëresults)
SELECT * FROM table_name
LIMIT 10;            -- first 10 rows
SELECT * FROM table_name
LIMIT 5, 10;         -- skip first 5, then next 10

-- distinct results (duplicate avoid)
SELECT DISTINCT col1 FROM table_name;
